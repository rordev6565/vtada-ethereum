def do_request(method, params = [], id = @id) #TODO: More error handling
  request_json = { :jsonrpc => @@jsonrpc,
             :method => method,
             :params => params,
             :id => @id
           }.to_json

  if @@debug
    puts "Request to " + @endpoint + ": " + request_json + "  ...."
  end

  response = HTTParty.post(@endpoint,
      :body => request_json,
      :headers => { 'Content-Type' => 'application/json' } )

  if @@debug
    puts "Response: " + response.to_s()
  end

  if response.bad_gateway?
    raise "Unable to connect to JSON-RPC endpont" + @endpoint
  end

  if !response.success?
    raise "JSON-RPC endpoint " + @endpoint + " returned http code " + response.code.to_s()
  end

  if response["error"]
    code = response["error"]["code"]
    message = response["error"]["message"]
    raise "In response to " + method + " request, JSON-RPC endpoint returned error #" + code.to_s() + ": " + message
  end
  response
end

def to_decimal(hex)
  if hex == nil
    return nil
  end
  if hex.is_a?(Integer)
    return hex
  end
  hex.to_i(16)
end

def to_hex(decimal)
  if decimal == nil

  end

    return decimal
  end
  decimal.to_s(16)
end
