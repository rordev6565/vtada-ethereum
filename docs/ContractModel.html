def initialize(abi)
  @contractMethods = Set.new
  for obj in abi
    if obj["type"] == "function" then
      inputs = obj["inputs"].map {|input| input["type"]}
      method = obj["name"] + "(" + inputs.join(",") + ")"
      @contractMethods.add method
    end
  end
end
